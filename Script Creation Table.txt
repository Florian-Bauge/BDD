CREATE TABLE client (
code_client integer PRIMARY KEY,
name varchar(255) NOT NULL,
Facebook varchar(255),
Instagram varchar(255),
Email varchar(255) NOT NULL,
Phone varchar(255),
id_membership integer NOT NULL,
point integer DEFAULT 0 
)ENGINE = InnoDB;

CREATE TABLE points (
id_point integer PRIMARY KEY,
point integer NOT NULL,
dateExp date,
code_client integer REFERENCES client
) ENGINE = InnoDB;

CREATE TABLE commande (
id_commande integer PRIMARY KEY,
fdelivery double NOT NULL,
total double,
statut varchar(255),
note varchar(255),
date date,
fservice double,
code_client integer REFERENCES client,
id_con integer REFERENCES concierge
)ENGINE = InnoDB;

CREATE TABLE envoie (
id_item integer REFERENCES Item,
id_livraison integer REFERENCES Livraison,
id_commande integer REFERENCES Commande,
Prix_remise double,
statut varchar(255),
quantité integer,
PRIMARY KEY(id_item, id_livraison, id_commande)
)ENGINE = InnoDB;

CREATE TABLE item (
id_item integer PRIMARY KEY,
prixachat double NOT NULL,
prixvente double NOT NULL,
nom varchar(255) NOT NULL,
statut varchar(255) NOT NULL,
id_membership integer DEFAULT 0,
stock integer NOT NULL
)ENGINE = InnoDB;

CREATE TABLE livraison (
id_delivery integer PRIMARY KEY,
numeroColis varchar(255),
dateVoulu date,
dateLivrée date,
DateExpédié date,
status varchar(255) NOT NULL,
id_adresse integer REFERENCES Adresse
)ENGINE = InnoDB;

CREATE TABLE adresse (
id_adresse integer PRIMARY KEY,
nrue varchar(255) NOT NULL,
rue varchar(255) NOT NULL,
codepostal integer  NOT NULL,
ville varchar(255) NOT NULL,
pays varchar(255) NOT NULL,
infoComp varchar(255),
code_client integer REFERENCES client
)ENGINE = InnoDB;

CREATE TABLE concierge (
id_con integer PRIMARY KEY,
nom varchar(255) NOT NULL,
prenom varchar(255) NOT NULL
)ENGINE = InnoDB;

CREATE TABLE facture (
id_fact integer PRIMARY KEY,
date date NOT NULL
)ENGINE = InnoDB;

CREATE TABLE paiement (
id_paiment integer PRIMARY KEY,
date date NOT NULL,
cout double NOT NULL,
id_transaction integer REFERENCES moyen
)ENGINE = InnoDB;

CREATE TABLE moyen (
id_transaction integer PRIMARY KEY,
nom varchar(255) NOT NULL
)ENGINE = InnoDB;

CREATE TABLE GrillePoint (
id_membership integer PRIMARY KEY,
nom varchar(255) NOT NULL,
minPoint integer NOT NULL,
maxPoint integer NOT NULL,
multiplication double NOT NULL
)ENGINE = InnoDB;
